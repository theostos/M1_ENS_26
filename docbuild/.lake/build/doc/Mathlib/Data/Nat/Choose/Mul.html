    theorem Nat.choose_mul_add {m n : ℕ} (hn : n ≠ 0) :
    (m * n + n).choose n = (m + 1) * (m * n + n - 1).choose (n - 1)

    theorem Nat.choose_mul_right {m n : ℕ} (hn : n ≠ 0) :
